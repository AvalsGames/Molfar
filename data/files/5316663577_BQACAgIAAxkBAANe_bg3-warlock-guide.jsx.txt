function SubBtn({ sc, selected, setSelected }) {
  const active = selected.id === sc.id;
  return (
    <button onClick={() => setSelected(sc)} style={{
      background: active ? `linear-gradient(135deg, ${sc.color}22, ${sc.accent}33)` : "rgba(255,255,255,0.03)",
      border: `1px solid ${active ? sc.accent : "rgba(255,255,255,0.1)"}`,
      borderRadius: 8, padding: "8px 4px", cursor: "pointer", textAlign: "center",
      boxShadow: active ? `0 0 14px ${sc.accent}44` : "none", transition: "all 0.15s"

      <div style={{ fontSize: 10, color: active ? sc.color : "#9a8a6a", fontWeight: active ? 600 : 400, lineHeight: 1.3 }}>{sc.name}</div>

  );
}
